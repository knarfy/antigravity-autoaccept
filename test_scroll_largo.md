```python
def algoritmo_complejo(datos, iteraciones):
    resultados = []
    for i in range(iteraciones):
        temporal = []
        for x in datos:
            if x % 2 == 0:
                temporal.append(x * i)
            else:
                temporal.append(x + i)
        
        # Simulación de un paso largo
        suma_total = sum(temporal)
        resultados.append(suma_total)
        
        if suma_total > 1000000:
            print(f"Límite alcanzado en la iteración {i}")
            break
            
    return resultados

# Ejecución
datos_prueba = list(range(100))
res = algoritmo_complejo(datos_prueba, 500)
print(f"Se obtuvieron {len(res)} resultados.")
```
